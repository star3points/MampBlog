CREATE TABLE `mypetdb`.`users` ( `id` INT NOT NULL AUTO_INCREMENT , `login` VARCHAR(40) NOT NULL , `password` VARCHAR(61) NOT NULL , PRIMARY KEY (`id`))

CREATE TABLE `mypetdb`.`comments` ( `id` INT NOT NULL AUTO_INCREMENT , `article_name` VARCHAR(100) NOT NULL , `user_login` VARCHAR(41) NOT NULL , `datetime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `comment` TEXT NOT NULL , PRIMARY KEY (`id`))
